    import math
    def pythagoreanTheorem(length_a, length_b):
        length_h = math.sqrt(pow(length_a,2) + pow(length_b,2))
        return length_h
    print(pythagoreanTheorem(2, 3))
    print(pythagoreanTheorem(3, 4))
    print(pythagoreanTheorem(4, 5))
